import discord
import os
from dotenv import load_dotenv
from system_core import ShareHouseSystem
import personas

# 設定読み込み
load_dotenv()
TOKEN = os.getenv('DISCORD_TOKEN')

# Discord接続準備
intents = discord.Intents.default()
intents.message_content = True
client = discord.Client(intents=intents)

# 世界の生成
house = ShareHouseSystem()

@client.event
async def on_ready():
    print(f'ログインしました: {client.user}')

@client.event
async def on_message(message):
    # 緊急メンテナンス（詰んだ時の救済処置）
    if message.content == "緊急メンテ":
        house.battery = 100.0
        house.dirt = 0.0   # 部屋もピカピカにする
        house.concepts = [] # 空気の重さもリセット
        
        await send_as_character(message.channel, "システム", "【システム】管理者権限による強制介入を確認……。\n生命維持装置、出力最大。環境浄化シーケンス完了。\n住人の意識レベル、正常値に戻りました。")
        return

    if message.author == client.user:
        return

    # 世界の状態を更新・取得
    status = house.get_status()
    
    # 返答を生成
    response = personas.get_response(message.content, status)
    
    # 送信
    await message.channel.send(response)

# 起動
client.run(TOKEN)
